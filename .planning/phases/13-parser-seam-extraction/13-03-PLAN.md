    func repeatASCII(s string, n int) string {
        if n <= 0 {
            return ""
        }
        out := make([]byte, 0, len(s)*n)
        for i := 0; i < n; i++ {
            out = append(out, s...)

        return string(out)

        t.Helper()
        path := filepath.Join("testdata", "parity-golden", name+".bin")

    func assertByteIdentical(t *testing.T, fixtureName string, encoded, golden []byte) {
        t.Helper()
        if len(encoded) != len(golden) {
            t.Fatalf("parity %s: byte length differs (encoded=%d golden=%d)",
                fixtureName, len(encoded), len(golden))
        }
        if !bytes.Equal(encoded, golden) {
            for i := range encoded {
                if encoded[i] != golden[i] {
                    t.Fatalf("parity %s: first diff at byte offset %d (encoded=0x%02x golden=0x%02x)",
                        fixtureName, i, encoded[i], golden[i])
                }
            }
        }
    }

    func buildAndEncode(t *testing.T, fx parityFixture) []byte {
        t.Helper()
        cfg := fx.Config()
        builder, err := NewBuilder(cfg, fx.NumRGs)
        if err != nil {
            t.Fatalf("NewBuilder for %s: %v", fx.Name, err)
        }
        for i, doc := range fx.JSONDocs {
            if err := builder.AddDocument(DocID(i), doc); err != nil {
                t.Fatalf("AddDocument[%d] for %s: %v", i, fx.Name, err)
            }
        }
        idx := builder.Finalize()
        encoded, err := Encode(idx)
        if err != nil {
            t.Fatalf("Encode for %s: %v", fx.Name, err)
        }
        return encoded
    }

    func TestParserParity_AuthoredFixtures(t *testing.T) {
        if !goldensPresent() {
            t.Skip("goldens not yet present in testdata/parity-golden/ — run Task 2 (parity_goldens_test.go under -tags regenerate_goldens) to generate them, then re-run this test")
        }
        for _, fx := range authoredParityFixtures() {
            fx := fx

                encoded := buildAndEncode(t, fx)
                golden := loadGolden(t, fx.Name)
                assertByteIdentical(t, fx.Name, encoded, golden)
            })
        }
    }

    // -----------------------------------------------------------------------------
    // D-05 dim #2 — gopter determinism: identical inputs → identical bytes
    // -----------------------------------------------------------------------------

    func TestParserParity_GopterByteIdentical(t *testing.T) {
        params := gopter.DefaultTestParameters()
        params.MinSuccessfulTests = 50
        properties := gopter.NewProperties(params)

        properties.Property("determinism across GenTestDocs", prop.ForAll(
            func(docs []TestDoc) bool {
                if len(docs) == 0 {
                    return true
                }
                a := encodeDocs(docs)
                b := encodeDocs(docs)
                return bytes.Equal(a, b)
            },
            GenTestDocs(25),
        ))

        properties.Property("determinism across GenTestDocsWithNulls", prop.ForAll(
            func(docs []TestDoc) bool {
                if len(docs) == 0 {
                    return true
                }
                a := encodeDocs(docs)
                b := encodeDocs(docs)
                return bytes.Equal(a, b)
            },
            GenTestDocsWithNulls(25),
        ))

        properties.Property("determinism across GenMixedTypeDocs", prop.ForAll(
            func(docs []TestDoc) bool {
                if len(docs) == 0 {
                    return true
                }
                a := encodeDocs(docs)
                b := encodeDocs(docs)
                return bytes.Equal(a, b)
            },
            GenMixedTypeDocs(25),
        ))

        properties.TestingRun(t)
    }

    // encodeDocs is a gopter helper: build + encode and return bytes. Returns
    // nil on internal failure so the property reports false rather than panicking.
    func encodeDocs(docs []TestDoc) []byte {
        if len(docs) == 0 {
            return nil
        }
        builder, err := NewBuilder(DefaultConfig(), len(docs))
        if err != nil {
            return nil
        }
        for i, doc := range docs {
            if err := builder.AddDocument(DocID(i), doc.JSON); err != nil {
                return nil
            }
        }
        idx := builder.Finalize()
        encoded, err := Encode(idx)
        if err != nil {
            return nil
        }
        return encoded
    }
